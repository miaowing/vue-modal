# vue-modal plugin

####init modal
```javascript
var vm = Vue.modal({
  target: element
  template: '<div>test</div>',
  data: {
      
  },
  autoDestroy: false
});
```

####open
```javascript
vm.open();
```

####close
```javascript
vm.close();
```

####destroy
```javascript
vm.destroy();
```

####update
```javascript
vm.udpate({
  
})
```
